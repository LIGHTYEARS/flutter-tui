# Plan 02: Command Palette and Remaining Shortcuts

<objective>

<action>
Modify `packages/flitter-amp/src/app.ts`:

1. Add import:
```typescript
import { CommandPalette } from './widgets/command-palette';
```

2. Add private state field to `AppStateWidget`:
```typescript
private showCommandPalette = false;
```

3. In the FocusScope `onKey` handler, add Ctrl+O handling:
```typescript
if (event.ctrlKey && event.key === 'o') {
  this.setState(() => { this.showCommandPalette = true; });
  return 'handled';
}
```

4. Modify the Escape handler to also dismiss command palette:
```typescript
if (event.key === 'Escape') {
  if (this.showCommandPalette) {
    this.setState(() => { this.showCommandPalette = false; });
    return 'handled';
  }
  if (appState.hasPendingPermission) {
    appState.resolvePermission(null);
    return 'handled';
  }
  return 'ignored';
}
```

5. In the build method, after the permission dialog check, add command palette overlay check:
```typescript
if (this.showCommandPalette) {
  return new Stack({
    children: [
      mainContent,
      new Positioned({
        top: 0, left: 0, right: 0, bottom: 0,
        child: new CommandPalette({
          onExecute: (command: string) => {
            this.setState(() => { this.showCommandPalette = false; });
            switch (command) {
              case 'clear':
                appState.conversation.clear();
                break;
              case 'toggle-tools':
                appState.conversation.toggleToolCalls();
                break;
              case 'toggle-thinking':
                this.toggleThinking(appState);
                break;
            }
          },
          onDismiss: () => {
            this.setState(() => { this.showCommandPalette = false; });
          },
        }),
      }),
    ],
  });
}
```

6. Add a `toggleThinking` helper method to `AppStateWidget`:
```typescript
private toggleThinking(appState: AppState): void {
  for (const item of appState.conversation.items) {
    if (item.type === 'thinking') {
      item.collapsed = !item.collapsed;
    }
  }
}
```

Note: Permission dialog takes priority over command palette (check hasPendingPermission first in build method).
